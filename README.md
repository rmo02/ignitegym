```bash
# Clone este repositório
$ git clone <https://github.com/rmo02/ignitegym.git>

# Acesse a pasta do projeto no terminal/cmd
$ cd ignitegym


# Instale as dependências
$ npm i

# Execute a aplicação em modo de desenvolvimento
$ expo start

#Acesse a pasta server - Back-end
$ cd server

# Instale as dependências
$ npm i

# Execute a aplicação em modo de desenvolvimento
$ npm run dev

# O Back-end será inciado na porta:3333 - acesse <http://localhost:3333>
```
